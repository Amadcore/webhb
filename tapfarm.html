from telegram.ext import Application, CommandHandler from Bot.config import BOT_TOKEN from Bot.commands import start, stats, add_invite, remove_user from Bot.database import init_db def main(): init_db() # Инициализируем базу данных (если таблицы не созданы, они будут созданы) app = Application.builder().token(BOT_TOKEN).build() app.add_handler(CommandHandler("start", start)) app.add_handler(CommandHandler("stats", stats)) app.add_handler(CommandHandler("add_invite", add_invite)) app.add_handler(CommandHandler("remove_user", remove_user)) print("Бот запущен! Ссылка: https://t.me/Hellbrush_web_bot") app.run_polling() if __name__ == '__main__': main()
